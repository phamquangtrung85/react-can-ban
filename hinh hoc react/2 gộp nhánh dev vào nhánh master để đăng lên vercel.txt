# 1. Chắc chắn bạn đang ở nhánh dev và đã commit/push mọi thứ
git checkout dev

cd tên dự án
git add .
git commit -m "nhập nội dung phiên làm việc"
git push origin dev

# 2. Chuyển sang nhánh master
git checkout master

# 3. Cập nhật master
git pull origin master

# 4. Hợp nhất dev vào master
git merge dev

(# Nếu có xung đột, giải quyết chúng: KHÔNG CÓ THÌ BỎ QUA PHẦN LỆNH NÀY

# 5. Đẩy master lên kho lưu trữ từ xa
git push origin master
